from sklearn.metrics import mean_squared_error

def BGD(X, y, w, num_of_epochs = 1000, learning_rate = 0.01):
    w = np.copy(w)
    num_samples = X.shape[0]

    print(mean_squared_error(y, X @ w))

    for epoch in range(num_of_epochs):
        grad = (2 / num_samples) * X.T @ (X @ w - y)
        w = w - learning_rate * grad
    #

    print(mean_squared_error(y, X @ w))
    return w
#

def BGD_errors_tracked(X, y, w, num_of_epochs = 1000, learning_rate = 0.01):
    w = np.copy(w)
    num_samples = X.shape[0]

    mean_squared_errors = [mean_squared_error(y, X @ w)]

    for epoch in range(num_of_epochs):
        grad = (2 / num_samples) * X.T @ (X @ w - y)
        w = w - learning_rate * grad

        mean_squared_errors.append(mean_squared_error(y, X @ w))
    #

    return w, mean_squared_errors
#

def MBGD(X, y, w, batch_size = 32, num_of_epochs = 1000, learning_rate = 0.01):
    w = np.copy(w)
    num_samples = X.shape[0]

    print(mean_squared_error(y, X @ w))

    for epoch in range(num_of_epochs):
        indices = np.random.permutation(num_samples)
        X_shuffled = X[indices]
        y_shuffled = y[indices]

        for start in range(0, num_samples, batch_size):
            X_batch = X_shuffled[start:start + batch_size]
            y_batch = y_shuffled[start:start + batch_size]
            current_batch_size = X_batch.shape[0]

            grad = (2 / current_batch_size) * X_batch.T @ (X_batch @ w - y_batch)
            w = w - learning_rate * grad
        #
    #

    print(mean_squared_error(y, X @ w))
    return w
#
